`confirm` callback: Uses `displayIntent` from `'../../intent/confirm-loop.js'` to show intent, then uses a helper `askQuestion(rl, prompt)` that creates a per-question AbortController:
```typescript
async function askQuestion(rl: Interface, prompt: string): Promise<string | null> {
  const ctrl = new AbortController();
  activeQuestionController = ctrl;
  try {
    return await rl.question(prompt, { signal: ctrl.signal });
  } catch (err) {
    if ((err as Error).name === 'AbortError') return null;
    throw err;
  } finally {
    activeQuestionController = null;
  }
}
```

The `confirm` callback implements a simplified confirm loop (do NOT call `confirmLoop` from confirm-loop.ts — it creates its own readline):
```typescript
const confirmCb: SessionCallbacks['confirm'] = async (intent, reparse) => {
  let current = intent;
  let attempts = 0;
  const maxRedirects = 3;
  while (attempts < maxRedirects) {
    displayIntent(current);
    const answer = await askQuestion(rl, pc.bold('  Proceed? [Y/n] '));
    if (answer === null) return null; // Ctrl+C
    if (answer === '' || answer.toLowerCase() === 'y') return current;
    if (answer.toLowerCase() !== 'n') {
      current = await reparse(answer);
      attempts++;
      continue;
    }
    attempts++;
    if (attempts >= maxRedirects) {
      console.log(pc.red('\n  Please try again with a clearer command'));
      return null;
    }
    const correction = await askQuestion(rl, '  Correction: ');
    if (correction === null) return null;
    current = await reparse(correction);
  }
  displayIntent(current);
  const finalAnswer = await askQuestion(rl, pc.bold('  Proceed? [Y/n] '));
  if (finalAnswer === null) return null;
  if (finalAnswer === '' || finalAnswer.toLowerCase() === 'y') return current;
  console.log(pc.red('\n  Please try again with a clearer command'));
  return null;
};
```

**Result block rendering:**

```typescript
function renderResultBlock(result: RetryResult, intent?: ResolvedIntent): void {
  const statusColor = result.finalStatus === 'success' ? pc.green
    : result.finalStatus === 'cancelled' ? pc.yellow
    : pc.red;
  const verifyResult = result.verificationResults.length > 0
    ? (result.verificationResults[result.verificationResults.length - 1].passed ? 'PASS' : 'FAIL')
    : 'N/A';
  const judgeVerdict = result.judgeResults && result.judgeResults.length > 0
    ? result.judgeResults[result.judgeResults.length - 1].verdict
    : 'N/A';

  console.log('');
  console.log(pc.dim('  ┌─────────────────────────────────────────┐'));
  console.log(`  │  ${pc.bold('Status:')}    ${statusColor(result.finalStatus.padEnd(28))}│`);
  console.log(`  │  ${pc.bold('Attempts:')}  ${String(result.attempts).padEnd(28)}│`);
  console.log(`  │  ${pc.bold('Verify:')}    ${verifyResult.padEnd(28)}│`);
  console.log(`  │  ${pc.bold('Judge:')}     ${judgeVerdict.padEnd(28)}│`);
  console.log(pc.dim('  └─────────────────────────────────────────┘'));
  console.log('');
}
```

**Main loop:**
```typescript
const state = createSessionState();
const registry = new ProjectRegistry();

while (true) {
  rl.setPrompt(getPrompt(state));
  const input = await askQuestion(rl, getPrompt(state));
  if (input === null) continue; // Ctrl+C at prompt — re-prompt

  // Create per-task AbortController
  activeTaskController = new AbortController();
  firstSigint = false;

  const callbacks: SessionCallbacks = {
    confirm: confirmCb,
    clarify: clarifyCb,
    getSignal: () => activeTaskController!.signal,
  };

  try {
    const output = await processInput(input, state, callbacks, registry);
    if (output.action === 'quit') {
      console.log(pc.dim('\n  Goodbye.\n'));
      rl.close();
      return;
    }
    if (output.result) {
      renderResultBlock(output.result, output.intent);
    } else if (output.result === null) {
      // User cancelled — just re-prompt silently
    }
  } catch (err) {
    if ((err as Error).name === 'AbortError' || activeTaskController.signal.aborted) {
      const forced = activeTaskController.signal.reason?.message === 'force';
      console.log(pc.yellow(`\n  Task cancelled${forced ? ' (forced)' : ''}.`));
    } else {
      console.error(pc.red(`\n  Error: ${(err as Error).message}`));
    }
  } finally {
    activeTaskController = null;
    firstSigint = false;
  }
}
```

In `src/cli/index.ts`, modify the `.action()` handler to add a REPL branch BEFORE the signal handler setup and BEFORE the legacy flag check.

The current flow is:
1. Create AbortController
2. Set up process SIGINT/SIGTERM handlers
3. If `input` -> one-shot
4. If no `--task-type` or `--repo` -> `program.help()`
5. Else -> legacy flag-based run

Change to:
1. **If `!input && !options.taskType` -> REPL path (no signal handlers needed)**
2. Create AbortController (only for one-shot/legacy paths)
3. Set up process SIGINT/SIGTERM handlers (only for one-shot/legacy paths)
4. If `input` -> one-shot
5. Else -> legacy flag-based run

Specifically, at the START of the action callback, add:

```typescript
// REPL path: no positional input and no --task-type flag
// Must be checked BEFORE installing process signal handlers.
// The REPL manages its own SIGINT via readline — process handlers would conflict.
if (!input && !options.taskType) {
  const { replCommand } = await import('./commands/repl.js');
  await replCommand();
  process.exit(0);
  return;
}
```

This block goes BEFORE the `const abortController = new AbortController();` line. This is critical because:
- The REPL owns all signal handling through `rl.on('SIGINT')` on the readline interface
- If `process.on('SIGINT', handleSignal)` is registered first, it will call `process.exit(130)` on the first Ctrl+C, killing the REPL instead of cancelling the task
- The dynamic import ensures the REPL module and nanospinner are only loaded when needed

Remove or keep the existing `program.help()` fallback — it is now unreachable for the no-args case but still serves as a catch for invalid flag combinations (e.g., `--repo` without `--task-type`). Actually, the existing check `if (!options.taskType || !options.repo)` already handles that. Since the REPL catches `!input && !options.taskType` first, the only remaining path to the legacy block requires both flags. The `program.help()` line at the original `if (!options.taskType || !options.repo)` now only triggers when `--task-type` is given without `--repo` (or vice versa), which is correct behavior.
